<script>
  // Function to fetch weather data and display it
  function fetchWeather() {
    const apiKey = 'your-api-key'; // Replace 'your-api-key' with your actual API key
    const apiUrl = `https://api.open-meteo.com/weather`;

    $.get(`${apiUrl}?latitude=YOUR_LATITUDE&longitude=YOUR_LONGITUDE&current_weather=true&timezone=auto&appid=${apiKey}`, function(data) {
      // Extract relevant weather information from the response
      const temperature = data.current_weather.temperature.value;
      const weatherDescription = data.current_weather.weather.description;
      
      // Create HTML elements to display weather information
      const weatherInfoElement = `
        <p>Temperature: ${temperature} Â°C</p>
        <p>Weather: ${weatherDescription}</p>
      `;
      
      // Append weather information to the div
      $('#weather-info').html(weatherInfoElement);
    });
  }

  // Call the fetchWeather function when the page loads
  $(document).ready(function() {
    fetchWeather();
  });
</script>
